#!/usr/bin/env python3
import argparse, zipfile, re, csv
from pathlib import Path

def extract_docx_text(docx_file: Path) -> str:
    with zipfile.ZipFile(docx_file, 'r') as z:
        xml = z.read('word/document.xml').decode('utf-8', errors='ignore')
    text = re.sub(r'<w:tab[^>]*?/>', '\t', xml)
    text = re.sub(r'</w:p>|<w:br[^>]*?/?>', '\n', text, flags=re.IGNORECASE)
    text = re.sub(r'<.*?>', '', text)
    text = text.replace('&amp;', '&').replace('&lt;', '<').replace('&gt;', '>')
    text = re.sub(r'\s+\n', '\n', text); text = re.sub(r'\n\s+', '\n', text)
    return text

def normalize(s: str) -> str:
    return re.sub(r'\s+', ' ', s).strip()

def parse_docx_qas(raw_text: str):
    q_header = re.compile(r'(?m)^\s*(\d{1,4})\s*[.)-:]\s*')
    matches = list(q_header.finditer(raw_text))
    for i, m in enumerate(matches):
        qid = int(m.group(1))
        start = m.start()
        end = matches[i+1].start() if i+1 < len(matches) else len(raw_text)
        block = raw_text[start:end]
        btxt = q_header.sub('', block, count=1).strip()
        choice_line_re = re.compile(r'(?m)^[ \t]*([A-F])[.)][ \t]+(.+)')
        first_choice = choice_line_re.search(btxt)
        if first_choice:
            stem = normalize(btxt[:first_choice.start()])
            choices_region = btxt[first_choice.start():]
        else:
            stem = normalize(btxt); choices_region = ""
        choices = {k:"" for k in "ABCDEF"}
        if choices_region:
            current = None; buf = []
            for line in choices_region.splitlines():
                ans_head = re.match(r'(?i)^\s*ANSWER:\s*([A-F])', line)
                if ans_head:
                    if current and buf:
                        choices[current] = normalize(" ".join(buf)); current=None; buf=[]
                m2 = choice_line_re.match(line)
                if m2:
                    if current and buf: choices[current] = normalize(" ".join(buf))
                    current = m2.group(1).upper(); buf = [m2.group(2)]
                else:
                    if current is not None: buf.append(line)
            if current and buf: choices[current] = normalize(" ".join(buf))
        ans_match = re.search(r'(?i)ANSWER:\s*([A-F])', block)
        correct = ans_match.group(1).upper() if ans_match else ""
        explanation = ""
        if ans_match:
            post = block[ans_match.end():]
            ref_pos = re.search(r'(?i)\bReference\b', post)
            next_q = q_header.search(post)
            cutoff = min([p.start() for p in [ref_pos, next_q] if p] or [len(post)])
            exp_raw = post[:cutoff]
            explanation = normalize(exp_raw.strip().split('\n')[0])[:500]
        yield qid, stem, choices, correct, explanation

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--docx", required=True)
    ap.add_argument("--year", type=int, required=True)
    ap.add_argument("--out", required=True)
    a = ap.parse_args()
    raw_text = extract_docx_text(Path(a.docx))
    rows = []
    for qid, stem, choices, correct, explanation in parse_docx_qas(raw_text):
        row = {"id": qid, "year": a.year, "stem": stem, "correct": correct,
               "explanation": explanation, "tags": "", "confidence": ""}
        for L in "ABCDEF": row[L] = choices.get(L, "")
        rows.append(row)
    import csv
    header = ["id","year","stem","A","B","C","D","E","F","correct","explanation","tags","confidence"]
    Path(a.out).parent.mkdir(parents=True, exist_ok=True)
    with open(a.out, "w", newline="", encoding="utf-8") as f:
        w = csv.DictWriter(f, fieldnames=header); w.writeheader()
        for r in sorted(rows, key=lambda x: int(x["id"])):
            w.writerow(r)
    print(f"Wrote {len(rows)} rows -> {a.out}")

if __name__ == "__main__":
    main()
